
# Usage examples

## count\_unique

``` r
library(miscR)

counted = count_unique(iris, 
                       subset_by = "Species", 
                       count_col = 1)
head(counted)
#>   Sepal.Length setosa versicolor virginica
#> 1          4.9      1          1         4
#> 2          5.6      1          5         0
#> 3          5.7      1          5         2
#> 4          5.8      3          3         1
#> 5          5.9      1          2         0
#> 6            6      2          4         0
species <- iris$Species[c(1,60)]
counted = count_unique(iris, 
                       subset_by = "Species", 
                       count_col = "Sepal.Length",
                       subset_values = species)
head(counted)
#>   Sepal.Length setosa versicolor
#> 1          4.9      1          4
#> 2            5      2          8
#> 3          5.1      1          8
#> 4          5.2      1          3
#> 5          5.4      1          5
#> 6          5.5      5          2
```
